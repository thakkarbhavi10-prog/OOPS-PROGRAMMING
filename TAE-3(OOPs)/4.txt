class Resort {
    int Rno, Days;
    String Name;
    double Charges;

    void Getinfo(int r, String n, double c, int d) {
        Rno = r;
        Name = n;
        Charges = c;
        Days = d;
    }

    double Compute() {
        double amt = Days * Charges;
        if (amt > 11000)
            amt = 1.02 * amt;
        return amt;
    }

    void DispInfo() {
        System.out.println(Rno + " " + Name + " " + Days + " " + Compute());
    }
}
