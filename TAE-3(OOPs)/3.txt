class CARRENTAL {
    int CarId;
    String CarType;
    float Rent;

    void GetCar(int id, String type) {
        CarId = id;
        CarType = type;
    }

    float GetRent() {
        if (CarType.equals("Small"))
            Rent = 1000;
        else if (CarType.equals("Van"))
            Rent = 800;
        else
            Rent = 2500;

        return Rent;
    }

    void ShowCar() {
        System.out.println(CarId + " " + CarType + " " + Rent);
    }
}
